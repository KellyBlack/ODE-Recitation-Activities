    \begin{subproblem}
      \item $\vec{u} = 3\vec{\imath} - 2\vec{\jmath} + 2\vec{k}$

  \item Show that the system of equations given by
    \begin{eqnarray}
      3x + 4y & = & 1, \\
      x + y   & = & 0.
    \end{eqnarray}
    is equivalent to the system given by
  \begin{eqnarray}
    \left[
      \begin{array}{rr}
        3 & 4 \\ 
        1 & 1
      \end{array}
    \right] 
    \left[
      \begin{array}{r}
        x \\
        y
      \end{array}
    \right]
    & = & 
    \left[
      \begin{array}{r}
        1 \\
        0
      \end{array}
    \right].
  \end{eqnarray}


  \iftoggle{solutions}{%

    The left hand side is 
    \begin{eqnarray*}
          \left[
      \begin{array}{rr}
        3 & 4 \\ 
        1 & 1
      \end{array}
    \right] 
    \left[
      \begin{array}{r}
        x \\
        y
      \end{array}
    \right]
    & = & 
    \left[
      \begin{array}{r}
        3x + 4y \\
        x + y
      \end{array}
    \right].
    \end{eqnarray*}

    This is the same as the left hand side of the equations in the
    problem statement.

  }

  \item Given a matrix,
    \begin{eqnarray}
      C & = &  \left[ \begin{array}{rr}
                         3 & 4 \\ 
                         1 & 1
                       \end{array} \right],
    \end{eqnarray}
    find the matrices, $A$ and $B$ that satisfy the following
    requirements.
    \begin{subproblem}
    \item Find a matrix, $A$, so that
      \begin{eqnarray}
        A * \left[
          \begin{array}{rr}
            3 & 4 \\ 
            1 & 1
          \end{array}
        \right] 
        & = & 
        \left[
          \begin{array}{rr}
             1 & 0 \\ 
             1 & 1
          \end{array}
        \right] 
      \end{eqnarray}
  
      \iftoggle{solutions}{%

        We need to find values of $a$ and $b$ so that
        \begin{eqnarray*}
          \left[
          \begin{array}{rr}
            a & b \\ 
            0 & 1
          \end{array}
        \right] 
          \left[
          \begin{array}{rr}
            3 & 4 \\ 
            1 & 1
          \end{array}
        \right]  & = & 
          \left[
          \begin{array}{rr}
            1 & 0 \\ 
            1 & 1
          \end{array}
        \right].
        \end{eqnarray*}

        The left hand side reduces to
        \begin{eqnarray*}
          \left[
          \begin{array}{rr}
            3a+b & 4a+b \\ 
            1 & 1
          \end{array}
          \right]
        \end{eqnarray*}

        Solving the system of equations
        \begin{eqnarray*}
          3a+b & = & 1, \\
          4a+b & = & 0,
        \end{eqnarray*}
        yields the values
        \begin{eqnarray*}
          a & = & -1, \\
          b & = & 4.
        \end{eqnarray*}

      }
    
      \vfill

    \item Find a matrix, $B$, so that
      \begin{eqnarray}
        B * A * \left[
          \begin{array}{rr}
            3 & 4 \\ 
            1 & 1
          \end{array}
        \right] 
        & = & 
        \left[
          \begin{array}{rr}
             1 & 0 \\ 
             0 & 1
          \end{array}
        \right] 
      \end{eqnarray}


      \iftoggle{solutions}{%

        We need to find values of $a$ and $b$ so that
        \begin{eqnarray*}
          \left[
          \begin{array}{rr}
            1 & 0 \\ 
            a & b
          \end{array}
        \right] 
          \left[
          \begin{array}{rr}
            1 & 0 \\ 
            1 & 1
          \end{array}
        \right]  & = & 
          \left[
          \begin{array}{rr}
            1 & 0 \\ 
            0 & 1
          \end{array}
        \right].
        \end{eqnarray*}

        The left hand side reduces to
        \begin{eqnarray*}
          \left[
          \begin{array}{rr}
            1 & 0 \\
            a+b & b 
          \end{array}
          \right]
        \end{eqnarray*}

        Solving the system of equations
        \begin{eqnarray*}
          a+b & = & 0, \\
            b & = & 1,
        \end{eqnarray*}
        yields the values
        \begin{eqnarray*}
          a & = & -1, \\
          b & = & 1.
        \end{eqnarray*}


      }

      \vfill

      \item What does this imply about the matrix $C$?

      \iftoggle{solutions}{%

        The inverse of the matrix is
        \begin{eqnarray*}
          B*A & = & 
          \left[
          \begin{array}{rr}
             -1 &  4 \\
              1 & -3 
          \end{array}
          \right].         
        \end{eqnarray*}

        }{\vspace*{3em}}
          

      \end{subproblem}
